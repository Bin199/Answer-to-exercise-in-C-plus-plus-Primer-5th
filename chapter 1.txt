int main()
{

        sum += val;
        ++val;
    }
    std::cout << "Sum of 50 to 100 inclusive is " << sum << std::endl;
    return 0;
}

1.17
#include<iostream>

int main()
{
    int currVal = 0, val = 0;
    if (std::cin >> currVal) {
        int cnt = 1;
        while (std::cin >> val) {
            if (val == currVal)
                ++cnt;
            else {
                std::cout << currVal << " occurs "
                          << cnt << " times " << std::endl;
                currVal = val;
                cnt = 1;
            }
        }
        std::cout << currVal << " occurs "
                  << cnt << " times " << std::endl;
    }
    return 0;
}

#include <iostream>

void print_fun(int m, int n){
    for (int i = m; i >= n; --i){
        std::cout << i << std::endl;
    }
}

int main()
{
    int m, n;
    std::cin >> m;
    std::cin >> n;
    if (m < n){
        print_fun(n, m);
    } else {
        print_fun(m, n);
    }
    return 0;
}

#include "Sales_item.h"

int main()
{
    Sales_item item;
    Sales_item sum;
    int cnt = 0;
    while (std::cin >> item) {
        if (cnt == 0)
            sum = item;
        else {
            sum += item;
        }
        ++cnt;
    }
    std::cout << "The sum is " << sum << std::endl;
    return 0;
}
